### 模板

```c++
// N是元素个数，K是每个元素最大的长度，int不超过32位。
const int N = 1e5 + 10, K = 32, M = K * N;
struct BDTree{
    // Binary dic tree
    int tr[M][2], idx, cnt[M];
    BDTree() {
        idx = 0; 
        memset(tr, 0, sizeof tr);
        memset(cnt, 0, sizeof cnt);
    }
    void add(int x) {
        int p = 0;
        for (int i = K - 1; ~i; --i) {
            int b = (x >> i) & 1;
            if (!tr[p][b]) tr[p][b] = ++idx;
            cnt[p = tr[p][b]]++;
        }
    }

    void del(int x) {
        int p = 0;
        for (int i = K - 1; ~i; --i) 
            cnt[p = tr[p][(x >> i) & 1]]--;
    }

    int query(int x) {
        int p = 0, ans = 0;
        for (int i = K - 1; ~i; --i) {
            int b = (x >> i) & 1;
            ans = ans << 1;
            if (cnt[tr[p][b]]) ans++, p = tr[p][b];
            else p = tr[p][!b];
        }
        return ans;
    }
};
```

### 最大异或对

[acwing 143](https://www.acwing.com/problem/content/145/)

在给定的 N 个整数 A1，A2…… AN 中选出两个进行 xor（异或）运算，得到的结果最大是多少？

**输入格式**
第一行输入一个整数 N。
第二行输入 N 个整数 A1～AN。

**输出格式**
输出一个整数表示答案。

**数据范围**
- 1 ≤ N ≤ 1e5,
- 0 ≤ Ai < 2^31

**输入样例**
```
3
1 2 3
```

**输出样例**
```
3
```

**分析**

二进制字典树， 对于每一个A_i, 先对其取反，设为x，则与x异或值最大的数为 数组中尽量和x高位相同的数，贪心选取即可。 

**时间复杂度**  O(n * k), k是二进制的长度， 一般不超过32

```c++
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 1e5 + 10, K = 32, M = K * N;
struct BDTree{
    ...
};

int a[N];
int main() {   
    int n, ans = 0; cin >> n;
    BDTree bt;
    for (int i = 0; i < n; i ++ ) {
        cin >> a[i];
        bt.add(a[i]);
    }
    for (int i = 0; i < n; i ++ ) 
        ans = max(ans, bt.query(~a[i]));
    cout << ans << endl;
    return 0;
}
```

### 区间最大异或对

[牛客练习赛81 B](https://ac.nowcoder.com/acm/contest/11171/B)

n 个正整数 a[1],a[2], ..., a[n], 询问m个问题，每个问题给一个区间 `[l,r]` 在区间 `[l,r]` 中选择两个数，使其异或值最大。如果只能选出一朵花请直接输出 0

**输入描述**
- 第一行两个整数 n, m （1 <= n, m <= 5 * 10^3）
- 第二行 n 个整数 a[1],a[2], ..., a[n] (1 <= a[i] <= 2^10)
- 接下来m行，每行两个整数 l,r, (1 <= l <= r <= n)

**输出描述**
- 输出 m 行，表示每个问题的答案。

```c++
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 5010, K = 12, M = K * N;
struct BDTree{
    ...
};
int a[N];
int main() {   
    int n, m, l, r;
    cin >> n >> m;
    for (int i = 0; i < n; ++i)  cin >> a[i];
    for (int i = 0; i < m; ++i) {
        cin >> l >> r;
        if (l == r) cout << 0 << endl;
        else {
            BDTree bt;
            for (int j = l - 1; j < r; ++j ) bt.add(a[j]);
            int ans = 0;
            for (int j = l - 1; j < r; ++j )
                ans = max(ans, bt.query(~a[j]));
            cout << ans << endl;
        }
    }
}
```

### 树上最大异或值路径

[acwing 144 最长异或值路径](https://www.acwing.com/problem/content/146/)

给定一个树，树上的边都具有权值。
树中一条路径的异或长度被定义为路径上所有边的权值的异或和
给定上述的具有 n 个节点的树，你能找到异或长度最大的路径吗？

**输入格式**
- 第一行包含整数 n，表示树的节点数目。
- 接下来 n−1 行，每行包括三个整数 u，v，w，表示节点 u 和节点 v 之间有一条边权重为 w。

**输出格式**
- 输出一个整数，表示异或长度最大的路径的最大异或和。

**数据范围**
- 1 <= n <= 100000
- 0 <= u, v < n
- 0 <= w < 2^31

**分析**

<br />
![](/images/posts/leetcode/trie.png)
<br />

```c++
#include <iostream>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

const int N = 1e5 + 10, K = 32, M = K * N;
struct BDTree{
    ...
};

int a[N], n;
struct Edge{
    int to, cost;
    Edge(int t, int c):to(t), cost(c){};
};
vector<Edge> G[N];

void dfs(int u, int fa, int sum) {
    a[u] = sum;
    for (auto v : G[u]) {
        if (v.to != fa) {
            dfs(v.to, u, sum ^ v.cost);
        }
    }
}

int main() {
    cin >> n;
    int u, v, w;
    for (int i = 0; i < n - 1; i ++ ) {
        cin >> u >> v >> w;
        G[u].push_back({v,w});
        G[v].push_back({u,w});
    }
    dfs(0, -1, 0);
    BDTree bt;
    for (int i = 0; i < n; i ++ ) bt.add(a[i]);
    
    int ans = 0;
    for (int i = 0; i < n; i ++ ) ans = max(ans, bt.query(~a[i]));
    cout << ans << endl;
}
```

### 统计异或值在范围内的数对有多少

[leetcode 1803](#https://leetcode-cn.com/problems/count-pairs-with-xor-in-a-range/)

给你一个整数数组 nums （下标 从 0 开始 计数）以及两个整数：low 和 high ，请返回 漂亮数对 的数目。

漂亮数对 是一个形如 (i, j) 的数对，其中 0 <= i < j < nums.length 且 low <= (nums[i] XOR nums[j]) <= high 。

**提示**
- 1 <= nums.length <= 2 * 10^4 
- 1 <= nums[i] <= 2 * 10^4
- 1 <= low <= high <= 2 * 10^4

**分析**

使用trie树， query(x, h) 统计与x异或的结果小于h的数的个数

```c++
class Solution {
public:
    int tr[20005*16][2], cnt[20005*16], idx=0;
    void insert(int x) {
        int p = 0;
        for (int i = 15; i >= 0; --i) {
            int b = x >> i & 1;
            if (!tr[p][b]) tr[p][b]=++idx;
            p = tr[p][b];
            cnt[p]++;
        }
    }
    // 查询与x异或的结果小于hi的数的个数
    int query(int x, int hi) {
        int res = 0, p = 0;
        for (int i = 15; i >= 0; --i) {
            int a = x>>i&1, b = hi>>i&1;
            if (b ==0) p = tr[p][a];
            else res += cnt[tr[p][a]], p = tr[p][1 - a];

            if (!p) return res; 
        }
        return res;
    }
    int countPairs(vector<int>& a, int l, int h) {
        int ans=0;
        for(int i:a){
            ans += query(i,h + 1) - query(i,l);
            insert(i);
        }
        return ans;
    }
};
```

### 长度不超过m的最大子数组异或和

[acwing 3485](https://www.acwing.com/problem/content/3488/)

给定一个非负整数数列 a，初始长度为 N。请在所有长度不超过 M 的连续子数组中，找出子数组异或和的最大值。

子数组的异或和即为子数组中所有元素按位异或得到的结果。 注意子数组可以为空

```c++
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 1e5 + 10, K = 31, M = K * N;
struct BDTree{
    ...
};

int n, m, a[N], s[M];

int main() {
    cin>>n>>m;
    for (int i = 0; i < n; i ++ ) cin >> a[i]; 
    BDTree bt;
    bt.add(s[0]);
    for (int i = 1; i <= n; i ++ ) s[i]=s[i-1]^a[i-1];
    int ans = 0;
    for (int i = 1; i <= n; i ++ )
    {
        if (i > m) bt.del(s[i - m - 1]);
        ans = max(ans, bt.query(~s[i]));
        bt.add(s[i]);
    }
    cout << ans << endl;
    
}
```

### 数对异或和

[atcoder abc147_d](https://atcoder.jp/contests/abc147/tasks/abc147_d)

给一个长度为n的数组，a[0], a[1], ... a[n-1].

求所有满足 `1<=i<j<=N` 的pair(i,j)，其a[i]^a[j]的和对1e9+7取模的结果。

**数据范围**

- 2 <= N <= 3e5
- 0 <= a[i] < 2**60
- a[i] 是整型

**分析**

由于采取异或操作时，各个bit互不影响，因此可以每个bit独立计算，
对于第k位，只有a[i]和a[j]在第k位不同，其异或值才为1，否则为0，
因此，第k位异或值为1的数对 数目等于 （n个数中第k位为1的数目）* (n个数中第k位为0的数目)
所以只需求出每一位中为1和为0的数量，时间复杂度为O(N*log(max(a[i])))

```c++
#include <bits/stdc++.h>
using namespace std;
const int P = 1e9 + 7;
int main() {
    int n;
    cin >> n;
    vector<long long> a(n);
    for (int i = 0; i < n; ++i)
        cin >> a[i];
    long long ans = 0;
    for (int k = 0; k < 60; ++k) {
        int c[2] = {};
        for (int i = 0; i < n; ++i)
            ++c[a[i] >> k & 1];
        ans = (ans + (1LL << k) % P * c[0] % P * c[1]) % P;
    }
    cout << ans << endl;
    return 0;
}
```

### 数对最短距离异或和

[atcoder abc201_e](https://atcoder.jp/contests/abc201/tasks/abc201_e)

一个包含n个点，n-1条无向边的无环图（树），第i条边连接 u[i] 和v[i]，具有权重w[i], 
对于顶点对(x,y)，定义
dist(x,y) = 从x到y最短路径上的权重异或值。

对于所有满足 `1<=i<j<=N` 的顶点对(i,j)，，求dist(i,j)的和对1e9+7取模的结果。

**数据范围**
- 2 <= N <= 2e5
- 1 <= u[i] < v[i]
- 0 <= w[i] < 2**60
- 输入图是棵树
- 所有数值都是整型

**分析**

首先化简dist(i,j), 随意选择一个节点作为根，设为 x, 设k是i,j的最近公共祖先，则dist(i,j)满足如下性质：
`dist(i,j) = dist(k, i) ^ dist(k, j)`

继续化简
```
dist(i, j) = dist(k, i) ^ dist(k, j)
           = dist(k, i) ^ dist(k, j) ^ dist(x, k) ^ dist(x, k)
           = (dist(x, k) ^ dist(k, i)) ^ (dist(x, k) ^ dist(k, j))
           = dist(x, i) ^ dist(x, j)
```


问题转化为 对于所有满足 `1<=i<j<=N` 的顶点对(i,j)，，求dist(x, i) ^ dist(x, j)的和对1e9+7取模的结果。

对于每一个dist(x, i)，我们可以在O(N)时间内使用BFS求出每一个节点与x的dist值，将其存到数组里，问题则转化为上一题。

```c++
#include<bits/stdc++.h>
using namespace std;
using ll = long long;
const ll mod = 1e9+7;

int main(){
    int N; cin >> N;
    vector<vector<ll>> edge(N), weight(N);
    for(int i=1; i<N; i++){
        ll u,v,w; cin >> u >> v >> w; u--, v--;
        edge[u].push_back(v);
        edge[v].push_back(u);
        weight[u].push_back(w);
        weight[v].push_back(w);
    }
    vector<ll> dist(N,-1);
    queue<int> que;
    que.push(0);
    dist[0] = 0;
    while(!que.empty()){
        int now = que.front(); que.pop();
        for(int i=0; i<edge[now].size(); i++){
            int next = edge[now][i];
            ll sum = dist[now]^weight[now][i];
            if(dist[next] == -1){
                dist[next] = sum;
                que.push(next);
            }
        }
    }
    ll ans = 0;
    for(int i=0; i<60; i++){
        vector<int> cnt(2);
        for(int j=0; j<N; j++) cnt[dist[j]>>i&1]++;
        ans += (1ll<<i)%mod*cnt[0]%mod*cnt[1];
        ans %= mod;
    }
    cout << ans << endl;
}
```

### 与数组中元素的最大异或值

[leetcode 1707](https://leetcode-cn.com/problems/maximum-xor-with-an-element-from-array/)

给你一个由非负整数组成的数组 nums 。另有一个查询数组 queries ，其中 queries[i] = [xi, mi] 。
第i个查询的答案是xi和任意数组中不超过mi的元素按位异或得到的最大值，

返回一个数组ans，ans.size() == querys.size()， 且ans[i]是第i个查询的答案。

```c++
const int N = 1e5 + 10, K = 31, M = K * N;
struct BDTree{
    ...
};
class Solution {
public:
    vector<int> maximizeXor(vector<int>& nums, vector<vector<int>>& qs) {
        BDTree bt;
        sort(nums.begin(), nums.end());
        for(int i = 0; i < qs.size(); ++i) qs[i].push_back(i);
        sort(qs.begin(), qs.end(),[&](const auto& a, const auto& b){return a[1] < b[1];});
        vector<int> ans(qs.size());
        int i = 0;
        for (auto e : qs){
            int x = e[0], m = e[1], qid = e[2];
            while(i < nums.size() && nums[i] <= m) {
                bt.add(nums[i]);
                i++;
            }
            if (!i) ans[qid] = -1;
            else ans[qid] = bt.query(~x);
        }
        return ans;
    }
};
```
